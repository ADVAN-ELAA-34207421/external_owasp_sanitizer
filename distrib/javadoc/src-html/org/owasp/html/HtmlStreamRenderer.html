040    /**
041     * Given a series of HTML tokens, writes valid, normalized HTML to the output.
042     * The output will have well-defined tag boundaries, but there may be orphaned
043     * or missing close and open tags.
044     * The result of two renderers can always be concatenated to produce a larger
045     * snippet of HTML, but if the first was called with
046     * {@code writeOpenTag("plaintext", ...)}, then any tags in the second will not
047     * be interpreted as tags in the concatenated version.
048     */
049    @TCB
050    @NotThreadSafe
051    public class HtmlStreamRenderer implements HtmlStreamEventReceiver {
052    
053      private final Appendable output;
054      private final Handler<? super IOException> ioExHandler;
055      private final Handler<? super String> badHtmlHandler;
056      private String lastTagOpened;
057      private StringBuilder pendingUnescaped;
058      private boolean open;
059    
060      /**
061       * Factory.
062       * @param output the buffer to which HTML is streamed.
063       * @param ioExHandler called with any exception raised by output.
064       * @param badHtmlHandler receives alerts when HTML cannot be rendered because
065       *    there is not valid HTML tree that results from that series of calls.
066       *    E.g. it is not possible to create an HTML {@code <style>} element whose
067       *    textual content is {@code "</style>"}.
068       */
069      public static HtmlStreamRenderer create(
070          @WillCloseWhenClosed Appendable output,
071          Handler<? super IOException> ioExHandler,
072          Handler<? super String> badHtmlHandler) {
073        if (output instanceof Closeable) {
074          return new CloseableHtmlStreamRenderer(
075              output, ioExHandler, badHtmlHandler);
076        } else {
077          return new HtmlStreamRenderer(output, ioExHandler, badHtmlHandler);
078        }
079      }
080    
081      /**
082       * Factory.
083       * @param output the buffer to which HTML is streamed.
084       * @param badHtmlHandler receives alerts when HTML cannot be rendered because
085       *    there is not valid HTML tree that results from that series of calls.
086       *    E.g. it is not possible to create an HTML {@code <style>} element whose
087       *    textual content is {@code "</style>"}.
088       */
089      public static HtmlStreamRenderer create(
090          StringBuilder output, Handler<? super String> badHtmlHandler) {
091        // Propagate since StringBuilder should not throw IOExceptions.
092        return create(output, Handler.PROPAGATE, badHtmlHandler);
093      }
094    
095      private HtmlStreamRenderer(
096          Appendable output, Handler<? super IOException> ioExHandler,
097          Handler<? super String> badHtmlHandler) {
098        this.output = output;
099        this.ioExHandler = ioExHandler;
100        this.badHtmlHandler = badHtmlHandler;
101      }
102    
103      /**
104       * Called when the series of calls make no sense.
105       * May be overridden to throw an unchecked throwable, to log, or to take some
106       * other action.
107       *
108       * @param message for human consumption.
109       * @param identifier an HTML identifier associated with the message.
110       */
111      private final void error(String message, CharSequence identifier) {
112        if (badHtmlHandler != Handler.DO_NOTHING) {   // Avoid string append.
113          badHtmlHandler.handle(message + " : " + identifier);
114        }
115      }
116    
117      public final void openDocument() throws IllegalStateException {
118        if (open) { throw new IllegalStateException(); }
119        open = true;
120      }
121    
122      public final void closeDocument() throws IllegalStateException {
123        if (!open) { throw new IllegalStateException(); }
124        if (pendingUnescaped != null) {
125          closeTag(lastTagOpened);
126        }
127        open = false;
128        if (output instanceof Flushable) {
129          try {
130            ((Flushable) output).flush();
131          } catch (IOException ex) {
132            ioExHandler.handle(ex);
133          }
134        }
135      }
136    
137      public final boolean isDocumentOpen() {
138        return open;
139      }
140    
141      public final void openTag(String elementName, List<String> attrs) {
142        try {
143          writeOpenTag(elementName, attrs);
144        } catch (IOException ex) {
145          ioExHandler.handle(ex);
146        }
147      }
148    
149      private void writeOpenTag(String elementName, List<? extends String> attrs)
150          throws IOException {
151        if (!open) { throw new IllegalStateException(); }
152        elementName = safeName(elementName);
153        if (!isValidHtmlName(elementName)) {
154          error("Invalid element name", elementName);
155          return;
156        }
157        if (pendingUnescaped != null) {
158          error("Tag content cannot appear inside CDATA element", elementName);
159          return;
160        }
161    
162        switch (HtmlTextEscapingMode.getModeForTag(elementName)) {
163          case CDATA_SOMETIMES:       
164          case CDATA:
165          case PLAIN_TEXT:
166            lastTagOpened = elementName;
167            pendingUnescaped = new StringBuilder();
168            break;
169          default:
170        }
171    
172        output.append('<').append(elementName);
173    
174        for (Iterator<? extends String> attrIt = attrs.iterator();
175             attrIt.hasNext();) {
176          String name = attrIt.next();
177          String value = attrIt.next();
178          name = HtmlLexer.canonicalName(name);
179          if (!isValidHtmlName(name)) {
180            error("Invalid attr name", name);
181            continue;
182          }
183          output.append(' ').append(name).append('=').append('"');
184          escapeHtmlOnto(value, output);
185          if (value.indexOf('`') != -1) {
186            // Apparently, in quirks mode, IE8 does a poor job producing innerHTML
187            // values.  Given
188            //     <div attr="``foo=bar">
189            // we encode &#96; but if JavaScript does:
190            //    nodeA.innerHTML = nodeB.innerHTML;
191            // and nodeB contains the DIV above, then IE8 will produce
192            //     <div attr=``foo=bar>
193            // as the value of nodeB.innerHTML and assign it to nodeA.
194            // IE8's HTML parser treats `` as a blank attribute value and foo=bar
195            // becomes a separate attribute.
196            // Adding a space at the end of the attribute prevents this by forcing
197            // IE8 to put double quotes around the attribute when computing
198            // nodeB.innerHTML.
199            output.append(' ');
200          }
201          output.append('"');
202        }
203    
204        output.append('>');
205      }
206    
207      public final void closeTag(String elementName) {
208        try {
209          writeCloseTag(safeName(elementName));
210        } catch (IOException ex) {
211          ioExHandler.handle(ex);
212        }
213      }
214    
215      private final void writeCloseTag(String elementName)
216          throws IOException {
217        if (!open) { throw new IllegalStateException(); }
218        elementName = HtmlLexer.canonicalName(elementName);
219        if (!isValidHtmlName(elementName)) {
220          error("Invalid element name", elementName);
221          return;
222        }
223    
224        if (pendingUnescaped != null) {
225          if (!lastTagOpened.equals(elementName)) {
226            error("Tag content cannot appear inside CDATA element", elementName);
227            return;
228          } else {
229            StringBuilder cdataContent = pendingUnescaped;
230            pendingUnescaped = null;
231            int problemIndex = checkHtmlCdataCloseable(lastTagOpened, cdataContent);
232            if (problemIndex == -1) {
233              output.append(cdataContent);
234            } else {
235              error(
236                  "Invalid CDATA text content",
237                  cdataContent.subSequence(
238                      problemIndex,
239                      Math.min(problemIndex + 10, cdataContent.length())));
240              // Still output the close tag.
241            }
242          }
243          if ("plaintext".equals(elementName)) { return; }
244        }
245        output.append("</").append(elementName).append(">");
246      }
247    
248      public final void text(String text) {
249        try {
250          writeText(text);
251        } catch (IOException ex) {
252          ioExHandler.handle(ex);
253        }
254      }
255    
256      private final void writeText(String text) throws IOException {
257        if (!open) { throw new IllegalStateException(); }
258        if (pendingUnescaped != null) {
259          pendingUnescaped.append(text.replaceAll("\0", ""));
260        } else {
261          escapeHtmlOnto(text, output);  // Works for RCDATA.
262        }
263      }
264    
265      private static int checkHtmlCdataCloseable(
266          String localName, StringBuilder sb) {
267        int escapingTextSpanStart = -1;
268        for (int i = 0, n = sb.length(); i < n; ++i) {
269          char ch = sb.charAt(i);
270          switch (ch) {
271            case '<':
272              if (i + 3 < n
273                  && '!' == sb.charAt(i + 1)
274                  && '-' == sb.charAt(i + 2)
275                  && '-' == sb.charAt(i + 3)) {
276                if (escapingTextSpanStart == -1) {
277                  escapingTextSpanStart = i;
278                } else {
279                  return i;
280                }
281              } else if (i + 1 + localName.length() < n
282                         && '/' == sb.charAt(i + 1)
283                         && Strings.regionMatchesIgnoreCase(
284                             sb, i + 2, localName, 0, localName.length())) {
285                // A close tag contained in the content.
286                if (escapingTextSpanStart < 0) {
287                  // We could try some recovery strategies here.
288                  // E.g. prepending "/<!--\n" to sb if "script".equals(localName)
289                  return i;
290                }
291                if (!"script".equals(localName)) {
292                  // Script tags are commonly included inside script tags.
293                  // <script><!--document.write('<script>f()</script>');--></script>
294                  // but this does not happen in other CDATA element types.
295                  // Actually allowing an end tag inside others is problematic.
296                  // Specifically,
297                  // <style><!--</style>-->/* foo */</style>
298                  // displays the text "/* foo */" on some browsers.
299                  return i;
300                }
301              }
302              break;
303            case '>':
304              // From the HTML5 spec:
305              //    The text in style, script, title, and textarea elements must not
306              //    have an escaping text span start that is not followed by an
307              //    escaping text span end.
308              // We look left since the HTML 5 spec allows the escaping text span
309              // end to share dashes with the start.
310              if (i >= 2 && '-' == sb.charAt(i - 1) && '-' == sb.charAt(i - 2)) {
311                if (escapingTextSpanStart < 0) { return i - 2; }
312                escapingTextSpanStart = -1;
313              }
314              break;
315          }
316        }
317        if (escapingTextSpanStart >= 0) {
318          // We could try recovery strategies here.
319          // E.g. appending "//-->" to the buffer if "script".equals(localName)
320          return escapingTextSpanStart;
321        }
322        return -1;
323      }

326      @VisibleForTesting
327      static boolean isValidHtmlName(String name) {
328        int n = name.length();
329        if (n == 0) { return false; }
330        if (n > 128) { return false; }
331        boolean isNamespaced = false;
332        for (int i = 0; i < n; ++i) {
333          char ch = name.charAt(i);
334          switch (ch) {
335            case ':':
336              if (isNamespaced) { return false; }
337              isNamespaced = true;
338              if (i == 0 || i + 1 == n) { return false; }
339              break;
340            case '-':
341              if (i == 0 || i + 1 == n) { return false; }
342              break;
343            default:
344              if (ch <= '9') {
345                if (i == 0 || ch < '0') { return false; }
346              } else if ('A' <= ch && ch <= 'z') {
347                if ('Z' < ch && ch < 'a') { return false; }
348              } else {
349                return false;
350              }
351              break;
352          }
353        }
354        return true;
355      }

357      /**
358       * Writes the HTML equivalent of the given plain text to output.
359       * For example, {@code escapeHtmlOnto("1 < 2", w)},
360       * is equivalent to {@code w.append("1 &lt; 2")} but possibly with fewer
361       * smaller appends.
362       */
363      static void escapeHtmlOnto(String plainText, Appendable output)
364          throws IOException {
365        int n = plainText.length();
366        int pos = 0;
367        for (int i = 0; i < n; ++i) {
368          char ch = plainText.charAt(i);
369          if (ch < REPLACEMENTS.length) {
370            String repl = REPLACEMENTS[ch];
371            if (repl != null) {
372              output.append(plainText, pos, i).append(repl);
373              pos = i + 1;
374            }
375          } else if (Character.isHighSurrogate(ch) && i + 1 < n) {
376            // Emit supplemental codepoints as entity so that they cannot
377            // be mis-encoded as UTF-8 of surrogates instead of UTF-8 proper
378            // and get involved in UTF-16/UCS-2 confusion.
379            char next = plainText.charAt(i + 1);
380            if (Character.isLowSurrogate(next)) {
381              int codepoint = Character.toCodePoint(ch, next);
382              output.append(plainText, pos, i);
383              appendNumericEntity(codepoint, output);
384              ++i;
385              pos = i + 1;
386            }
387            // All orphaned surrogates are rendered raw.
388          } else if (ch >= 0xff00) {
389            // Is a control character or possible full-width version of a
390            // special character.
391            output.append(plainText, pos, i);
392            appendNumericEntity(ch, output);
393            pos = i + 1;
394          }
395        }
396        output.append(plainText, pos, n);
397      }
398    
399      /** Maps ASCII chars that need to be encoded to an equivalent HTML entity. */
400      static final String[] REPLACEMENTS = new String[0x61];
401      static {
402        REPLACEMENTS[0] = "";                          // Elide.
403        for (int i = 1; i < ' '; ++i) {
404          if (i == '\n' || i == '\r') { continue; }
405          REPLACEMENTS[i] = "&#" + i + ";";
406        }
407        REPLACEMENTS['"'] = "&#" + ((int) '"') + ";";  // Attribute delimiter.
408        REPLACEMENTS['&'] = "&amp;";                   // HTML special.
409        REPLACEMENTS['\''] = "&#" + ((int) '\'') + ";";// Attribute delimiter.
410        REPLACEMENTS['+'] = "&#" + ((int) '+') + ";";  // UTF-7 special.
411        REPLACEMENTS['<'] = "&lt;";                    // HTML special.
412        REPLACEMENTS['='] = "&#" + ((int) '=') + ";";  // Special in attributes.
413        REPLACEMENTS['>'] = "&gt;";                    // HTML special.
414        REPLACEMENTS['@'] = "&#" + ((int) '@') + ";";  // Conditional compilation.
415        REPLACEMENTS['`'] = "&#" + ((int) '`') + ";";  // Attribute delimiter.
416      }
417    
418      static void appendNumericEntity(int codepoint, Appendable output)
419         throws IOException {
420        if (codepoint < 100) {
421          // TODO: is this dead code due to REPLACEMENTS above.
422          output.append("&#");
423          if (codepoint < 10) {
424            output.append((char) ('0' + codepoint));
425          } else {
426            output.append((char) ('0' + (codepoint / 10)));
427            output.append((char) ('0' + (codepoint % 10)));
428          }
429          output.append(";");
430        } else {
431          int nDigits = (codepoint < 0x1000
432                         ? codepoint < 0x100 ? 2 : 3
433                         : (codepoint < 0x10000 ? 4
434                            : codepoint < 0x100000 ? 5 : 6));
435          output.append("&#x");
436          for (int digit = nDigits; --digit >= 0;) {
437            int hexDigit = (codepoint >>> (digit << 2)) & 0xf;
438            output.append(HEX_NUMERAL[hexDigit]);
439          }
440          output.append(";");
441        }
442      }
443    
444      private static final char[] HEX_NUMERAL = {
445        '0', '1', '2', '3', '4', '5', '6', '7',
446        '8', '9', 'a', 'b', 'c', 'd', 'e', 'f',
447      };

449      /**
450       * Canonicalizes the element name and possibly substitutes an alternative
451       * that has more consistent semantics.
452       */ 
453      static String safeName(String elementName) {
454        elementName = HtmlLexer.canonicalName(elementName);
455    
456        // Substitute a reliably non-raw-text element for raw-text and
457        // plain-text elements.
458        switch (elementName.length()) {
459          case 3:
460            if ("xmp".equals(elementName)) { return "pre"; }
461            break;
462          case 7:
463            if ("listing".equals(elementName)) { return "pre"; }
464            break;
465          case 9:
466            if ("plaintext".equals(elementName)) { return "pre"; }
467            break;
468        }
469        return elementName;
470      }

472      static class CloseableHtmlStreamRenderer extends HtmlStreamRenderer
473          implements Closeable {
474        private final Closeable closeable;
475    
476        CloseableHtmlStreamRenderer(
477            @WillCloseWhenClosed
478            Appendable output, Handler<? super IOException> errorHandler,
479            Handler<? super String> badHtmlHandler) {
480          super(output, errorHandler, badHtmlHandler);
481          this.closeable = (Closeable) output;
482        }
483    
484        public void close() throws IOException {
485          if (isDocumentOpen()) { closeDocument(); }
486          closeable.close();
487        }
488      }
