029    package org.owasp.html;
030    
031    import java.util.List;
032    
033    import com.google.common.collect.ImmutableMap;
034    import com.google.common.collect.Lists;
035    
036    /**
037     * Wraps an HTML stream event receiver to fill in missing close tags.
038     * If the balancer is given the HTML {@code <p>1<p>2}, the wrapped receiver will
039     * see events equivalent to {@code <p>1</p><p>2</p>}.
040     *
041     * @author Mike Samuel <mikesamuel@gmail.com>
042     */
043    @TCB
044    public class TagBalancingHtmlStreamEventReceiver
045        implements HtmlStreamEventReceiver {
046      private final HtmlStreamEventReceiver underlying;
047      private final List<String> openElements = Lists.newArrayList();
048    
049      public TagBalancingHtmlStreamEventReceiver(
050          HtmlStreamEventReceiver underlying) {
051        this.underlying = underlying;
052      }
053    
054      public void openDocument() {
055        underlying.openDocument();
056      }
057    
058      public void closeDocument() {
059        while (!openElements.isEmpty()) {
060          closeTag(openElements.get(openElements.size() - 1));
061        }
062        underlying.closeDocument();
063      }
064    
065      public void openTag(String elementName, List<String> attrs) {
066        String canonElementName = HtmlLexer.canonicalName(elementName);
067        String optionalEndTagPartition = END_TAG_PARTITIONS.get(canonElementName);
068        if (optionalEndTagPartition != null) {
069          int n = openElements.size();
070          while (--n >= 0) {
071            if (!optionalEndTagPartition.equals(
072                    END_TAG_PARTITIONS.get(openElements.get(n)))) {
073              break;
074            }
075            underlying.closeTag(openElements.remove(n));
076          }
077        }
078        if (HtmlTextEscapingMode.VOID != HtmlTextEscapingMode.getModeForTag(
079                canonElementName)) {
080          openElements.add(canonElementName);
081        }
082        underlying.openTag(elementName, attrs);
083      }
084    
085      public void closeTag(String elementName) {
086        String canonElementName = HtmlLexer.canonicalName(elementName);
087        int index = openElements.lastIndexOf(canonElementName);
088        if (index < 0) { return; }  // Don't close unopened tags.
089        int last = openElements.size();
090        while (--last > index) {
091          String unclosedElementName = openElements.remove(last);
092          underlying.closeTag(unclosedElementName);
093        }
094        openElements.remove(index);
095        underlying.closeTag(elementName);
096      }
097    
098      public void text(String text) {
099        underlying.text(text);
100      }
101    
102    
103      private static final ImmutableMap<String, String> END_TAG_PARTITIONS
104          = ImmutableMap.<String, String>builder()
105          .put("body", "body")
106          .put("colgroup", "colgroup")
107          .put("dd", "dd")
108          .put("dt", "dd")
109          .put("head", "body")
110          .put("li", "li")
111          .put("option", "option")
112          .put("p", "p")
113          .put("tbody", "tbody")
114          .put("td", "td")
115          .put("tfoot", "tbody")
116          .put("th", "td")
117          .put("thead", "tbody")
118          .put("tr", "tr")
119          .build();
120    }
